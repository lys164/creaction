# creaction → arca-i18n 同步 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 给 creaction 增加「导入POPOP」手动推送能力：把本地角色（人设+封面）、INS 帖子、落地页单向写入 arca-i18n 后端，不改动现有本地生成/落盘逻辑。

**Architecture:** 新增两个隔离模块——`app/arca_client.py`（arca HTTP 客户端：JWT、TOS 上传、建角色异步轮询、发帖）和 `app/arca_mapping.py`（本地 record → arca 请求体的纯映射函数），由 `app/arca_sync.py` 编排单角色同步并把 arca 返回的 id 回写本地实现幂等。服务端加 `POST /api/arca/sync`，复用现有 `app/tasks.py` 后台任务系统；前端加按钮。

**Tech Stack:** Python 3、FastAPI、requests、PyJWT（本地签 token）、tos（Volcengine TOS SDK，上传图片/HTML）、pytest（新引入测试框架）。

## Global Constraints

- 不修改任何现有生成逻辑（`pipeline.py` 的 create/generate 函数、`chat.py`、`landing.py` 的生成部分）；只**新增**文件、只在 `config.py`/`server.py`/`web/` 做加法式改动。
- arca 建角色异步：`POST /character/create` → `{task_id}`，轮询 `POST /task/get_status`，`status ∈ processing|ready|failed`，`ready` 后 `result`（JSON 串）解析出 `character_id`。
- JWT：HS256，claim 恰为 `{"uid": <ARCA_UID>, "exp": <epoch秒>}`，密钥 `ARCA_ACCESS_SECRET`；请求头 `Authorization: Bearer <token>`（**必须带 Bearer 前缀**）。默认 30 天过期。
- 所有 arca 写请求带业务头：`X-Language`（角色语言 zh|ja|ko|en）、`X-Region`、`X-App-Version`（来自 config，占位可空）。
- 幂等：本地 record 存 `arca_character_id`；帖子存 `post_id → arca_post_id` 映射；已同步则跳过（除非 force）。建角色请求带 `Idempotency-Key` 头。
- 枚举硬约束：`source="character"`；`gender ∈ male|female|other`；`post.visibility ∈ 1|2|3`（默认 1）；`UserUploadImage.image_type="aigc"`；`StorageObject.object_type="image"`；`OpeningPrologueData.output_type="text"`。
- 环境变量默认全部留占位空值，缺失时同步接口返回明确错误而非崩溃。
- batch-resilient：单角色/单帖失败不中断整批，错误落到任务结果里。

"""本地 record → arca 请求体的纯映射（无副作用、无 IO，便于单测）。"""
import json

_MALE = ("男", "male", "남")
_FEMALE = ("女", "female", "여")

# 已被直接映射的 persona 键，不再重复塞进 customized_settings
_MAPPED_KEYS = {"name", "profile", "gender", "species", "personality", "opening"}


def normalize_gender(value: str) -> str:
    v = (value or "").strip().lower()
    if any(t in v for t in (t.lower() for t in _FEMALE)):
        return "female"
    if any(t in v for t in (t.lower() for t in _MALE)):
        return "male"
    return "other"


def _stringify(value) -> str:
    if isinstance(value, str):
        return value
    return json.dumps(value, ensure_ascii=False)


def persona_to_character_form(persona: dict,
                              images: list[dict] | None = None,
                              landing_url: str | None = None) -> dict:
    persona = persona or {}
    form: dict = {}
    if persona.get("name"):
        form["name"] = persona["name"]
    if persona.get("profile"):
        form["profile"] = persona["profile"]
    if persona.get("gender"):
        form["gender"] = normalize_gender(persona["gender"])
    if persona.get("species"):
        form["species"] = _stringify(persona["species"])
    pers = persona.get("personality")
    if isinstance(pers, dict) and pers.get("summary"):
        form["disposition"] = pers["summary"]
    elif isinstance(pers, str) and pers:
        form["disposition"] = pers

    opening = persona.get("opening") or {}
    msgs = opening.get("messages") if isinstance(opening, dict) else None
    if msgs:
        form["opening_prologue"] = [
            {"text": _stringify(m), "output_type": "text"} for m in msgs if m
        ]

    # 兜底：其余字段（appearance/backstory/family/... 及完整 personality）保真进 map
    extra: dict[str, str] = {}
    for k, v in persona.items():
        if k in _MAPPED_KEYS or v in (None, "", [], {}):
            continue
        extra[k] = _stringify(v)
    if isinstance(pers, dict):  # personality 完整体也保真一份
        extra["personality"] = _stringify(pers)
    if extra:
        form["customized_settings"] = extra

    if images:
        form["images"] = images
    if landing_url:
        form["landing_page_url"] = landing_url
    return form
```

"""arca-i18n HTTP 客户端：JWT / TOS 上传 / 建角色(异步) / 发帖。"""
import time

import jwt as pyjwt
import requests

from . import config


class ArcaError(Exception):
    pass


def get_token() -> str:
    """返回裸 JWT（不含 Bearer 前缀）。"""
    if config.ARCA_JWT_MODE == "endpoint":
        if not config.ARCA_BASE_URL:
            raise ArcaError("ARCA_BASE_URL 未配置")
        r = requests.post(
            f"{config.ARCA_BASE_URL}/internal/tool/gen_jwt_token",
            json={"uid": config.ARCA_UID, "expires_in": config.ARCA_JWT_EXPIRES},
            timeout=15,
        )
        r.raise_for_status()
        data = (r.json() or {}).get("data") or {}
        tok = data.get("jwt_token")
        if not tok:
            raise ArcaError(f"gen_jwt_token 未返回 token: {r.text[:200]}")
        return tok
    # local 自签
    if not config.ARCA_ACCESS_SECRET:
        raise ArcaError("ARCA_ACCESS_SECRET 未配置（本地自签模式必需）")
    if not config.ARCA_UID:
        raise ArcaError("ARCA_UID 未配置")
    payload = {
        "uid": config.ARCA_UID,
        "exp": int(time.time()) + config.ARCA_JWT_EXPIRES,
    }
    return pyjwt.encode(payload, config.ARCA_ACCESS_SECRET, algorithm="HS256")


def auth_header() -> dict:
    return {"Authorization": f"Bearer {get_token()}"}
```

import json as _json


def _data(resp) -> dict:
    """取 arca 统一响应壳 {code,msg,data}；code!=0 抛业务错误。"""
    resp.raise_for_status()
    body = resp.json() or {}
    if body.get("code", 0) not in (0, None):
        raise ArcaError(f"arca 业务错误 code={body.get('code')} msg={body.get('msg')}")
    return body.get("data") or {}


def _headers(lang: str, idempotency_key: str | None = None) -> dict:
    h = auth_header()
    h["Content-Type"] = "application/json"
    h["X-Language"] = lang or "zh"
    if config.ARCA_REGION:
        h["X-Region"] = config.ARCA_REGION
    if config.ARCA_APP_VERSION:
        h["X-App-Version"] = config.ARCA_APP_VERSION
    if idempotency_key:
        h["Idempotency-Key"] = idempotency_key
    return h


def _get_task_status(task_id: str, lang: str) -> dict:
    r = requests.post(
        f"{config.ARCA_BASE_URL}/task/get_status",
        json={"task_id": task_id}, headers=_headers(lang), timeout=30,
    )
    return _data(r)


def create_character(form: dict, lang: str, idempotency_key: str | None = None,
                     poll_interval: float = 3.0, timeout: float = 300.0) -> str:
    if not config.ARCA_BASE_URL:
        raise ArcaError("ARCA_BASE_URL 未配置")
    r = requests.post(
        f"{config.ARCA_BASE_URL}/character/create",
        json={"character_create_form": form, "source": "character"},
        headers=_headers(lang, idempotency_key), timeout=60,
    )
    task_id = _data(r).get("task_id")
    if not task_id:
        raise ArcaError(f"create 未返回 task_id: {r.text[:200]}")

    deadline = time.time() + timeout
    while True:
        st = _get_task_status(task_id, lang)
        status = st.get("status")
        if status == "ready":
            result = st.get("result") or "{}"
            cid = (_json.loads(result) if isinstance(result, str) else result).get("character_id")
            if not cid:
                raise ArcaError(f"任务成功但无 character_id: {result[:200]}")
            return cid
        if status == "failed":
            raise ArcaError(f"建角色失败 code={st.get('error_code')} msg={st.get('error_message')}")
        if time.time() > deadline:
            raise ArcaError(f"建角色轮询超时 task_id={task_id}")
        if poll_interval:
            time.sleep(poll_interval)
```

def _tos_put_object(bucket, key, content, token, ak, sk, region, endpoint, content_type):
    """用 Volcengine TOS SDK PUT 一个对象。隔离成函数便于测试打桩。"""
    import tos
    client = tos.TosClientV2(ak, sk, endpoint, region, security_token=token)
    client.put_object(bucket, key, content=content, content_type=content_type)


def tos_upload(data: bytes, object_key: str, content_type: str,
               lang: str, public: bool = False) -> dict:
    if not config.ARCA_BASE_URL:
        raise ArcaError("ARCA_BASE_URL 未配置")
    r = requests.post(
        f"{config.ARCA_BASE_URL}/file/tos_credential",
        json={"use_public": public, "expires_in": 3600},
        headers=_headers(lang), timeout=30,
    )
    cred = _data(r)
    bucket = (config.ARCA_TOS_BUCKET_PUBLIC if public and config.ARCA_TOS_BUCKET_PUBLIC
              else cred.get("bucket"))
    endpoint = cred.get("endpoint")
    _tos_put_object(
        bucket, object_key, data,
        cred.get("session_token"), cred.get("access_key_id"),
        cred.get("secret_access_key"), cred.get("region"), endpoint,
        content_type,
    )
    host = endpoint.replace("https://", "").replace("http://", "")
    url = f"https://{bucket}.{host}/{object_key}"
    return {"bucket_name": bucket, "object_key": object_key,
            "object_type": "image", "url": url}


def create_post(character_id: str, content: str, image_objs: list[dict],
                lang: str, visibility: int | None = None) -> str:
    images = [
        {"image_type": "aigc", "is_main_pic": i == 0, "media": obj}
        for i, obj in enumerate(image_objs or [])
    ]
    payload = {
        "character_id": character_id,
        "content": content or "",
        "images": images,
        "visibility": visibility if visibility is not None else config.ARCA_POST_VISIBILITY,
    }
    r = requests.post(
        f"{config.ARCA_BASE_URL}/post/create",
        json=payload, headers=_headers(lang), timeout=60,
    )
    pid = _data(r).get("post_id")
    if not pid:
        raise ArcaError(f"create_post 未返回 post_id: {r.text[:200]}")
    return pid
```

> 注：`tos_upload` 的 URL 拼法与 `object_type`（图片 image / HTML 文本）在联调时按 CDN 实际域名核对；HTML 传公有桶时 content_type 用 `text/html`，object_type 仍传 `image` 不影响（该字段仅 arca 内部分类，落地页只用返回 url）。

```python
"""编排单个角色的 arca 同步：封面→(落地页)→建角色→发帖，并回写本地 id 实现幂等。"""
from pathlib import Path

from . import arca_client, arca_mapping, config, pipeline


def _latest_posts(char_id: str) -> list[dict]:
    """取该角色最近一批 INS 帖子（有则用 ig，退回普通批次的最新一批）。"""
    ig = pipeline.load_latest_ig(char_id)
    if ig and ig.get("posts"):
        return ig["posts"]
    return []


def _content_for_lang(content, lang: str) -> str:
    if isinstance(content, dict):
        return content.get(lang) or next((v for v in content.values() if v), "")
    return content or ""


def _upload_cover(record: dict, lang: str) -> list[dict]:
    cover = record.get("cover") or {}
    lp = cover.get("local_path")
    if lp and Path(lp).exists():
        data = Path(lp).read_bytes()
        key = f"creaction/{record['char_id']}/cover.png"
        return [arca_client.tos_upload(data, key, "image/png", lang)]
    return []


def _upload_landing(record: dict, lang: str) -> str | None:
    page = pipeline.load_latest_landing(record["char_id"])
    html = (page or {}).get("html_filled")
    if not html:
        return None
    key = f"creaction/{record['char_id']}/landing.html"
    obj = arca_client.tos_upload(html.encode("utf-8"), key, "text/html", lang, public=True)
    return obj.get("url")


def sync_character(char_id: str, *, force: bool = False,
                   sync_landing: bool | None = None, progress=None) -> dict:
    record = pipeline.load_character(char_id)
    lang = record.get("lang", config.LANGUAGES[0])
    result = {"char_id": char_id, "arca_character_id": record.get("arca_character_id"),
              "posts": [], "landing_url": None, "skipped": False, "errors": []}

    do_landing = config.ARCA_SYNC_LANDING if sync_landing is None else sync_landing

    # 1) 建角色（幂等：已同步则跳过）
    if record.get("arca_character_id") and not force:
        result["skipped"] = True
    else:
        images = _upload_cover(record, lang)
        landing_url = None
        if do_landing:
            try:
                landing_url = _upload_landing(record, lang)
                result["landing_url"] = landing_url
            except Exception as e:  # noqa: BLE001
                result["errors"].append(f"landing 上传失败: {e}")
        form = arca_mapping.persona_to_character_form(
            record.get("persona", {}), images=images, landing_url=landing_url)
        cid = arca_client.create_character(
            form, lang=lang, idempotency_key=f"creaction-{char_id}")
        record["arca_character_id"] = cid
        result["arca_character_id"] = cid
        pipeline.save_character(record)

    cid = result["arca_character_id"]
    if not cid:
        return result

    # 2) 发帖（逐条 resilient；已同步过的 post 跳过）
    synced = record.setdefault("arca_post_ids", {})
    for post in _latest_posts(char_id):
        pid = post.get("post_id")
        if not pid or (pid in synced and not force):
            if pid in synced:
                result["posts"].append({"post_id": pid, "arca_post_id": synced[pid]})
            continue
        try:
            content = _content_for_lang(post.get("content"), lang)
            img = post.get("image") or {}
            image_objs = []
            lp = img.get("local_path")
            if lp and Path(lp).exists():
                obj = arca_client.tos_upload(
                    Path(lp).read_bytes(),
                    f"creaction/{char_id}/post_{pid}.png", "image/png", lang)
                image_objs = [obj]
            arca_pid = arca_client.create_post(cid, content, image_objs, lang)
            synced[pid] = arca_pid
            result["posts"].append({"post_id": pid, "arca_post_id": arca_pid})
        except Exception as e:  # noqa: BLE001
            result["errors"].append(f"帖子 {pid} 同步失败: {e}")

    pipeline.save_character(record)
    return result
```

> 注：`pipeline.load_latest_ig` / `load_latest_landing` 已存在（见 `app/pipeline.py`）。若某角色无 ig 批次，`_latest_posts` 返回空、只同步人设。
